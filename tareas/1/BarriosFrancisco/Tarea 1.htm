void crearArreglo(int* arreglo,int size){

              int r, i;

              for (i=0;i<size;i++){

                            r = rand()%1000;            *Se puede usar información del tiempo del sistema para generar los números aleatorios

                            arreglo[i] = r;

              }

}

void printArrayCC(int* arreglo, int size){
